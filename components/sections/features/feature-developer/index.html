import { BenefitsAPI } from "@kota/sdk";

const api = new BenefitsAPI({
  apiKey: process.env.KOTA_API_KEY
});

const policy = await api.enrol({
  employeeId: "emp_abc123",
  product: "health_ie"
});
from kota import BenefitsAPI

api = BenefitsAPI(api_key=os.environ["KOTA_API_KEY"])

policy = api.enrol(
  employee_id="emp_abc123",
  product="health_ie"
)

print(policy.status)  # active
curl -X POST \
